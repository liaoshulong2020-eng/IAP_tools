## 核心源文件详解

### 1. main.c - 系统入口和主循环

**文件位置:** `main/main.c`

**关键内容:**

```c
// ========== 主函数入口 ==========
int main(void)
{
    // 1. 时钟配置 (180MHz)
    SystemClock_Config();

    // 2. 底层库初始化
    LL_Init();

    // 3. 延时20ms等待稳定
    delayms(20);

    // 4. 系统初始化
    sys_init();

    // 5. 主循环
    while(1)
    {
        // Modbus IAP通信任务
        modbus_iap_task();

        // IAP处理 (超时检查, Flash验证, APP跳转)
        iap_task();

        // 系统维护 (喂狗)
        sys_task();

        // 通信接口轮询
        #if USE_UART
            uart1_tx_poll();  // UART发送轮询
        #elif USE_CAN
            cans_rx_poll();   // CAN接收轮询
        #endif
    }
}
```

**理解要点:**
- main() 是程序的总控制中心
- 初始化顺序很重要
- 主循环是事件驱动的
- 三个主要任务协调运行

```c
// ========== IAP初始化 ==========
void iap_init(void)
{
    iap_flag = false;    // 未进入IAP
    time_cnt = 0;        // 超时计数
    jump_flag = false;   // 跳转标志
}

// ========== IAP任务 (主循环调用) ==========
void iap_task(void)
{
    // 已进入IAP模式，只等待命令
    if (iap_flag)
        return;

    // UART模式: 周期性发送进入IAP请求
    #if USE_UART
        request_enter_iap();
    #endif

    // 超时检查 (20000次循环 ≈ 400ms)
    if (time_cnt++ >= 20000)
    {
        // 验证Flash
        if (iap_flash_verify())
        {
            // 验证通过，准备跳转
            LL_IWDG_DeInit();
            LL_TMR_DeInit();
            jump_to_app();  // 不返回
        }
        else
        {
            // 验证失败，重新计时
            time_cnt = 0;
        }
    }
}

// ========== Flash验证 ==========
bool iap_flash_verify(void)
{
    // 从参数区读取app_size和crc32
    ulong app_size = *(ulong*)ARG_BASE_ADDR;
    ulong app_crc = *(ulong*)(ARG_BASE_ADDR + 4);

    // 检查app_size有效性
    if (app_size == 0 || app_size > APP_MAX_SIZE)
        return false;

    // 计算应用程序的CRC32
    ulong calc_crc = 0xFFFFFFFF;
    crc32_update(&calc_crc, (void*)APP_BASE_ADDR, app_size);

    // 比较CRC
    return (calc_crc == app_crc);
}

// ========== IAP命令分发 ==========
static void iap_pkt_decode(iap_pkt_t *pkt)
{
    switch (pkt->cmd)
    {
        case 0: cmd_exit_iap(pkt);      // 退出IAP
        case 1: cmd_enter_iap(pkt);     // 进入IAP
        case 2: cmd_read_flash(pkt);    // 读Flash
        case 3: cmd_write_flash(pkt);   // 写Flash
        case 4: cmd_write_checksum(pkt);// 写校验和
    }
}

// ========== Flash编程 ==========
static bool flash_program(ulong addr, const void *buff, ulong size)
{
    // 计算扇区索引
    ulong sec_idx = (addr - FLASH_BASE_ADDR) / SECTOR_SIZE;

    // 如果是扇区起始地址，先擦除
    if ((addr % SECTOR_SIZE) == 0)
    {
        if (!LL_EFLASH_EraseSector(sec_idx))
            return false;
    }

    // 编程Flash
    if (!LL_EFLASH_Program(addr, buff, size))
        return false;

    // 验证数据
    if (!LL_EFLASH_Verify(addr, buff, size))
        return false;

    return true;
}

// ========== 跳转到应用程序 ==========
static void jump_to_app(void)
{
    typedef void (*pFunction)(void);

    uint32_t app_addr = APP_BASE_ADDR;
    uint32_t app_stack = *(uint32_t*)app_addr;
    pFunction app_entry = (pFunction)(*(uint32_t*)(app_addr + 4));

    __disable_irq();      // 禁用中断
    __set_MSP(app_stack); // 设置堆栈指针
    app_entry();          // 跳转到APP (不返回)
}
```

**学习重点:**
- IAP命令的完整处理流程
- Flash编程的关键步骤
- CRC校验和验证逻辑
- 应用程序的跳转方式

```c
// ========== 接收字节处理 (状态机) ==========
void modbuss_recv_byte(uchar data)
{
    static uchar recv_state = 0;
    static uchar recv_cnt = 0;
    static uchar recv_buff[256];

    if (recv_state == 0)
    {
        // STATE 0: 等待地址和功能码
        recv_buff[recv_cnt++] = data;

        if (recv_cnt >= 2)
        {
            recv_state = 1;
            recv_cnt = 0;

            // 根据功能码确定需要接收的字节数
            uchar fno = recv_buff[1];
            // ...
        }
    }
    else if (recv_state == 1)
    {
        // STATE 1: 根据功能码接收数据
        recv_buff[recv_cnt++] = data;

        // 检查是否接收完整
        if (recv_cnt >= expected_size)
        {
            recv_state = 2;
            pkt_flag = true;  // 标记数据包完整
        }
    }
}

// ========== 数据包解码 ==========
static void pkt_decode(void)
{
    // CRC16校验
    ushort calc_crc = crc16_update(recv_buff, recv_size - 2);
    ushort recv_crc = (recv_buff[recv_size-1] << 8) | recv_buff[recv_size-2];

    if (calc_crc != recv_crc)
        return;  // CRC失败，丢弃

    // 地址匹配检查
    if (recv_buff[0] != from_addr)
        return;  // 地址不匹配

    // 解析功能码并调用回调
    uchar fno = recv_buff[1];
    if (fno == FNO_IAP)
    {
        // IAP命令，调用IAP回调
        on_iap(&recv_buff[2], recv_size - 4);
    }
}
```

**学习重点:**
- Modbus接收状态机的实现
- CRC16校验的应用
- 数据包的完整性检查

```c
// CRC32查表法实现
static const uint32_t crc32_table[256] = {
    0x00000000, 0x04C11DB7, 0x09823B6E, 0x0D4326D9, ...
};

void crc32_update(uint32_t *crc, const void *buff, uint32_t size)
{
    const uint8_t *p = (const uint8_t *)buff;
    uint32_t crc_val = *crc;

    for (uint32_t i = 0; i < size; i++)
    {
        uint8_t tbl_idx = (crc_val ^ p[i]) & 0xFF;
        crc_val = (crc_val >> 8) ^ crc32_table[tbl_idx];
    }

    *crc = crc_val;
}
```

**特点:**
- 使用查表法，速度快
- CRC32标准算法(IEEE 802.3)
- 用于Flash数据验证

```c
// 完整的IAP写Flash流程
static void cmd_write_flash(iap_pkt_t *pkt)
{
    // 1. 提取命令参数
    ulong addr = pkt->addr;        // Flash地址
    ushort len = pkt->len;         // 数据长度
    ushort size = pkt->size;       // 实际数据长度
    uchar *data = pkt->data;       // 数据缓冲区

    // 2. 验证参数
    if (size != len || size > 256 || (addr % 8) != 0)
    {
        pkt->len = 0;  // 返回失败
        pkt->size = 0;
        return;
    }

    // 3. 调用Flash编程
    bool result = flash_program(addr, data, size);

    // 4. 返回结果
    if (result)
    {
        pkt->size = 0;  // size=0表示成功
    }
    else
    {
        pkt->len = 0;   // len=0表示失败
        pkt->size = 0;
    }
}
```

### 2. Modbus接收和CRC校验

```c
// 完整的数据包接收和校验
void modbuss_task(void)
{
    if (!pkt_flag)
        return;

    // 1. CRC16校验
    ushort recv_size = recv_cnt;
    ushort calc_crc = 0xFFFF;

    for (int i = 0; i < recv_size - 2; i++)
    {
        calc_crc = crc16_update(&calc_crc, recv_buff[i]);
    }

    // 接收的CRC (小端序)
    ushort recv_crc = (recv_buff[recv_size-1] << 8) |
                      recv_buff[recv_size-2];

    if (calc_crc != recv_crc)
    {
        pkt_flag = false;
        return;  // CRC错误，丢弃
    }

    // 2. 地址检查
    if (recv_buff[0] != MODBUS_LOCAL_ADDR)
    {
        pkt_flag = false;
        return;  // 地址不匹配
    }

    // 3. 功能码检查和处理
    uchar fno = recv_buff[1];
    if (fno == 0x00)  // IAP功能码
    {
        // 调用IAP处理
        on_iap_cmd(&recv_buff[2]);
    }

    pkt_flag = false;
}
```

### 4. Flash验证和CRC计算

```c
// 完整的Flash验证流程
bool verify_app_in_flash(void)
{
    // 1. 从参数区读取应用信息
    uint32_t app_size = *(uint32_t*)ARG_BASE_ADDR;
    uint32_t stored_crc = *(uint32_t*)(ARG_BASE_ADDR + 4);

    // 2. 检查app_size有效性
    if (app_size == 0xFFFFFFFF || app_size == 0)
        return false;

    if (app_size > APP_MAX_SIZE)
        return false;

    // 3. 计算实际CRC32
    uint32_t calc_crc = 0xFFFFFFFF;

    // 按256字节块计算
    for (uint32_t i = 0; i < app_size; i += 256)
    {
        uint32_t block_size = (app_size - i > 256) ? 256 : (app_size - i);
        uint8_t *addr = (uint8_t*)(APP_BASE_ADDR + i);

        crc32_update(&calc_crc, addr, block_size);
    }

    // 4. 比较CRC
    if (calc_crc != stored_crc)
    {
        // 验证失败
        return false;
    }

    // 5. 验证成功
    return true;
}
```

### 5. 完整的IAP更新时序

```c
// 上位机视角的完整IAP更新
void iap_upgrade_process(void)
{
    // STEP 1: 进入IAP模式
    send_modbus_command(cmd=1, data=0x01);  // 进入IAP
    wait_response();

    // STEP 2: 循环发送固件数据
    for (int block = 0; block < total_blocks; block++)
    {
        uint32_t addr = APP_BASE_ADDR + (block * 256);
        uint8_t data[256] = {...};  // 固件数据

        send_modbus_command(cmd=3, addr=addr, data=data, size=256);
        wait_response();

        // 显示进度
        printf("Progress: %d/%d\n", block + 1, total_blocks);
    }

    // STEP 3: 写入校验信息
    uint32_t app_size = total_blocks * 256;
    uint32_t app_crc = calculate_crc32(firmware_data, app_size);

    uint8_t checksum_data[8] = {
        (app_size >> 24) & 0xFF,
        (app_size >> 16) & 0xFF,
        (app_size >> 8) & 0xFF,
        app_size & 0xFF,
        (app_crc >> 24) & 0xFF,
        (app_crc >> 16) & 0xFF,
        (app_crc >> 8) & 0xFF,
        app_crc & 0xFF
    };

    send_modbus_command(cmd=4, data=checksum_data);
    wait_response();

    // STEP 4: 退出IAP模式并验证
    send_modbus_command(cmd=0);  // 退出IAP
    response = wait_response();

    if (response->cmd == 0x0000)
    {
        printf("Upgrade successful!\n");
        // BootLoader会在下次循环跳转到APP
    }
    else if (response->cmd == 0xFFFF)
    {
        printf("CRC verification failed!\n");
        // 升级失败，可以重试
    }
}
```
